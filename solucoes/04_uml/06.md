## Pergunta

6 - Cria um projeto no Visual Studio que respeite o seguinte diagrama UML. O
projeto deve ter funcionalidade mínima para demonstrar as relações indicadas no
diagrama.

![uml06](../../img/uml06.png)

## Soluções

### Solução 1

**Player**

```cs
public class Player
{
    public int ID { get; }
    public string Name { get; }
    private double health, shield;

    public Player(int id, string name, double health, double shield)
    {
        ID = id;
        Name = name;
        this.health = health;
        this.shield = shield;
    }

    public void PrintPlayerName()
    {
        Console.WriteLine("A player, called {0}.", Name);
    }
}
```

**Map**

```cs
public abstract class Map
{
    public string Name { get; set; }
    protected string Filename { get; set; }

    public Player player1, player2, player3;

    public Map(string name, string filename)
    {
        Name = name;
        Filename = filename;
    }

    public void PrintPlayerNames()
    {
        player1.PrintPlayerName();
        player2.PrintPlayerName();
        player3.PrintPlayerName();
    }
}
```

**DungeonMap**

```cs
public class DungeonMap : Map
{
    public int NumberOfDungeons { get; }

    public DungeonMap(string name, string filename, int nOfDungeons)
        : base(name, filename)
    {
        Name = name;
        Filename = filename;
        NumberOfDungeons = nOfDungeons;
        player1 = new Player(1, "Dungeoneer John", 100, 100);
        player2 = new Player(2, "Dungeoneer Maria", 50, 50);
        player3 = new Player(3, "Dungeoneer Adam", 50, 0);
    }
}
```

**OpenWorldMap**

```cs
public class OpenWorldMap : Map
{
    public float Width { get; }
    public float Height { get; }

    public OpenWorldMap(string name, string filename, int w, int h)
        : base(name, filename)
    {
        Name = name;
        Filename = filename;
        Width = w;
        Height = h;
        player1 = new Player(1, "Adventurer John", 100, 100);
        player2 = new Player(2, "Adventurer Maria", 50, 50);
        player3 = new Player(3, "Adventurer Adam", 50, 0);
    }
}
```

**Game**

```cs
public class Game
{
    internal static int gamesPlayed;
    private int numberOfPlayers;

    private Map dMap, owMap;

    internal void GameLoop()
    {
        dMap = new DungeonMap("Cool Dungeon", "dungeonMapFileName", 3);
        owMap = new OpenWorldMap("Cool Open World", "owMapFileName", 20,
            20);

        gamesPlayed = 1;
        numberOfPlayers = 3;

        Console.WriteLine("Games played: {0}.\nWith {1} player(s).",
            gamesPlayed, numberOfPlayers);
        Console.Write("In the dungeon map, called {0}, there is...\n",
            dMap.Name);
        dMap.PrintPlayerNames();
        Console.Write("\n");
        Console.Write("In the open world map, called {0}, there is...\n",
            owMap.Name);
        owMap.PrintPlayerNames();
    }
}
```

**Program**

```cs
internal class Program
{
    static void Main(string[] args)
    {
        Game game = new Game();
        game.GameLoop();
    }
}
```
